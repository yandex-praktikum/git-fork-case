ssh -T git@github.com		подключение к гиту

git init			Инициализация репозитория
git remote add origin <url>	Привязка локального репозитория к удаленному
git remote -v			Проверка, что репозитории связались
git push -u origin main		В первый раз загрузи все коммиты из локального репозитория в удалённый с названием origin

git add <файл>			Подготовка файла к коммиту
git add --all			Подготовка всех файлов с изменениями и новых к коммиту
git add .			Подготовка к коммиту текущей папки и всех ее файлов
git commit -m "Комментарий"	Сделать коммит с комментарием
git push			Добавить изменения в удаленный репозиторий

git restore --staged <file>	Убрать файл из staging
git restore --staged .		Убрать текущую папку из staging
git reset --hard <commit hash>	Откатить до этого коммита
git restore <file>		Откатить изменения в файле

git diff			Сравнение закомиченной версии с измененной
git diff --staged		Показывает изменения в staged-файлах
git diff a9928ab 11bada1 	Выведи разницу между двумя коммитами
git log				Подробная история коммитов
git log --oneline		Краткая история изменений (логи)

git status			Текущее состояние репозитория
.gitignore			Создать файл и вписать в него игноры для отслеживания
git status --ignored		Выводит и игнорируемые файлы

git commit --amend --no-edit	Добавь изменения к последнему коммиту и оставь сообщение прежним
git commit --amend -m "Новое" 	Измени сообщение к последнему коммиту на Новое
